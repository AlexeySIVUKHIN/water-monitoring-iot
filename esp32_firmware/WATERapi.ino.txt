#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <OneWire.h>
#include <DallasTemperature.h>

// ========== НАСТРОЙКИ Wi-Fi ==========
const char* ssid = "...";        // ЗАМЕНИТЕ НА СВОЁ
const char* password = "...";    // ЗАМЕНИТЕ НА СВОЙ

// ========== АДРЕС БОЕВОГО СЕРВЕРА ==========
String serverName = "http://publicwater.ru/api/measurement/";

// ========== ИДЕНТИФИКАЦИЯ УСТРОЙСТВА ==========
const char* mac = "94:51:dc:30:44:1c";      // MAC вашей ESP32
const char* deviceId = "esp32_river_01";     // Уникальное имя датчика

// ========== ПИНЫ ДАТЧИКОВ ==========
#define TDS_POWER 17      // Питание TDS (GPIO17)
#define TDS_SENSOR 36     // Сигнал TDS (GPIO36/VP)
#define TURBIDITY_SENSOR 34  // Сигнал мутности (GPIO34)
#define NTC_PIN 39        // NTC термистор (GPIO39/VN)
#define ONE_WIRE_BUS 4    // DS18B20 (GPIO4) - запасной вариант

// ========== ПАРАМЕТРЫ NTC ==========
const float R0 = 10000.0;        // 10кОм при 25°C
const float BETA = 3950.0;       // BETA-коэффициент
const float R_BALANCE = 10000.0; // Резистор 10кОм

// ========== ПЕРЕМЕННЫЕ ==========
float temperature = 0.0;
int tdsValue = 0;
float turbidityVoltage = 0.0;

// ========== ГЕОГРАФИЧЕСКИЕ КООРДИНАТЫ ==========
float latitude = 57.019943;   // Широта (можно заменить на свои)
float longitude = 40.933426;  // Долгота (можно заменить на свои)

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("=================================");
  Serial.println("?? ЗАПУСК ДАТЧИКОВ");
  Serial.println("=================================");
  
  // Настройка пинов
  pinMode(TDS_POWER, OUTPUT);
  digitalWrite(TDS_POWER, LOW);  // TDS выключен
  
  // Настройка АЦП
  analogReadResolution(12);
  
  // Подключение к Wi-Fi
  WiFi.begin(ssid, password);
  Serial.print("Подключение к Wi-Fi");
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n? Wi-Fi подключен!");
    Serial.print("IP адрес ESP32: ");
    Serial.println(WiFi.localIP());
    Serial.print("Адрес сервера: ");
    Serial.println(serverName);
  } else {
    Serial.println("\n? Ошибка подключения к Wi-Fi");
  }
  Serial.println("=================================");
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    
    Serial.println("\n?? НОВОЕ ИЗМЕРЕНИЕ:");
    
    // Чтение температуры с NTC
    readNTC();
    
    // Чтение TDS
    readTDS();
    
    // Чтение мутности
    readTurbidity();
    
    // Отправка на сервер
    sendToServer();
    
  } else {
    Serial.println("? Wi-Fi отключен, пробуем переподключиться...");
    WiFi.reconnect();
  }
  
  // Задержка между измерениями (5 минут для экономии батареи)
  delay(300000); // 300000 мс = 5 минут
}

// ========== ФУНКЦИЯ ЧТЕНИЯ NTC ==========
void readNTC() {
  int raw = analogRead(NTC_PIN);
  
  if (raw == 0 || raw == 4095) {
    Serial.println("?? Ошибка NTC!");
    temperature = 0;
    return;
  }
  
  float voltage = raw * (3.3 / 4095.0);
  float resistance = R_BALANCE * (3.3 / voltage - 1);
  
  // Формула Стейнхарта-Харта
  float steinhart = log(resistance / R0) / BETA;
  steinhart += 1.0 / 298.15;
  steinhart = 1.0 / steinhart;
  temperature = steinhart - 273.15;
  
  Serial.print("??? Температура (NTC): ");
  Serial.print(temperature, 1);
  Serial.println(" °C");
}

// ========== ФУНКЦИЯ ЧТЕНИЯ TDS ==========
void readTDS() {
  // Включаем питание TDS
  digitalWrite(TDS_POWER, HIGH);
  delay(100);  // Ждём стабилизации
  
  // Усреднённое чтение (10 измерений)
  int sum = 0;
  const int samples = 10;
  for (int i = 0; i < samples; i++) {
    sum += analogRead(TDS_SENSOR);
    delay(10);
  }
  int raw = sum / samples;
  
  // Выключаем питание TDS (бережём электроды)
  digitalWrite(TDS_POWER, LOW);
  
  float voltage = raw * (3.3 / 4095.0);
  
  // Температурная компенсация
  if (temperature > 0) {
    voltage = voltage / (1.0 + 0.02 * (temperature - 25.0));
  }
  
  tdsValue = (int)(voltage * 500);  // Коэффициент 500 (можно откалибровать точнее)
  
  Serial.print("?? TDS: ");
  Serial.print(tdsValue);
  Serial.println(" ppm");
}

// ========== ФУНКЦИЯ ЧТЕНИЯ МУТНОСТИ ==========
void readTurbidity() {
  // Усреднённое чтение
  int sum = 0;
  const int samples = 10;
  for (int i = 0; i < samples; i++) {
    sum += analogRead(TURBIDITY_SENSOR);
    delay(10);
  }
  int raw = sum / samples;
  
  turbidityVoltage = raw * (3.3 / 4095.0);
  
  Serial.print("??? Мутность: ");
  Serial.print(turbidityVoltage, 3);
  Serial.println(" В");
}

// ========== ФУНКЦИЯ ОТПРАВКИ НА СЕРВЕР ==========
void sendToServer() {
  HTTPClient http;
  
  Serial.print("Отправка на сервер: ");
  Serial.println(serverName);
  
  http.begin(serverName);
  http.addHeader("Content-Type", "application/json");
  
  // Создаём JSON
  StaticJsonDocument<256> doc;
  
  doc["mac"] = mac;
  doc["temperature"] = temperature;
  doc["tds"] = tdsValue;
  doc["turbidity"] = turbidityVoltage;
  doc["signal"] = WiFi.RSSI();
  doc["latitude"] = latitude;
  doc["longitude"] = longitude;
  
  String jsonString;
  serializeJson(doc, jsonString);
  
  Serial.println("JSON: " + jsonString);
  
  int httpResponseCode = http.POST(jsonString);
  
  if (httpResponseCode > 0) {
    Serial.print("? Код ответа: ");
    Serial.println(httpResponseCode);
    
    if (httpResponseCode == 201) {
      String response = http.getString();
      Serial.println("Ответ сервера: " + response);
    }
  } else {
    Serial.print("? Ошибка отправки: ");
    Serial.println(http.errorToString(httpResponseCode).c_str());
  }
  
  http.end();
}
